25void Player::tick(long long elapsedTimeInMilliseconds) {
26
27 // Calcul de la distance parcourue par le player, selon sa vitesse et le temps écoulé.
28 QPointF playerDistance = elapsedTimeInMilliseconds * m_playerVelocity / 1000;
29 // Positionne la bounding box du player à sa prochaine position.
30 QRectF nextRect = this->globalBoundingBox().translated(playerDistance);
31 //Récupère tous les sprites de la scène
32 auto collidingSprites = this->parentScene()->collidingSprites(nextRect);
33 // Supprimer le sprite lui-même, qui collisionne toujours avec sa boundingbox
34 collidingSprites.removeAll(this);
35
36 //trie les bonus des obstacles
37 Sprite* bonus = nullptr;
38 for (Sprite* sprite : collidingSprites) {
39 if (sprite->data(0).toString() == "bonus") {
40 bonus = sprite;
41 }
42 }
43
44 bool collision = !collidingSprites.isEmpty();
45
46 // Si la prochaine position reste dans les limites de la scène, la balle
47 // y est positionnée. Sinon, elle reste sur place.
48 if (this->parentScene()->isInsideScene(nextRect)) {
49 this->setPos(this->pos() + playerDistance);
50 }
51
52 //Si le player collisionne avec un bonus, alors le bonus est supprimé
53 if (bonus != nullptr) {
54 bonus->deleteLater();
55 emit onBonusHit();
56 //Sinon le player meurt
57 } else if(collision && playerAlive){
58 playerAlive = false;
59 emit onplayerDestroyed();
60 }
61}

132void GameCanvas::keyPressed(QKeyEvent* pKeyEvent) {
133 // Supprimer ce premier test si la répétition de touche doit être signalée.
134 if (pKeyEvent->isAutoRepeat())
135 pKeyEvent->ignore();
136 else {
137 m_pGameCore->keyPressed(pKeyEvent->key());
138
139 if (pKeyEvent->modifiers()==(Qt::ShiftModifier|Qt::ControlModifier)) {
140 switch (pKeyEvent->key()) {
141 case Qt::Key_I:
142 if (m_pDetailedInfosItem)
143 m_pDetailedInfosItem->setVisible(!m_pDetailedInfosItem->isVisible());
144 break;
145 case Qt::Key_P:
146 m_tickTimer.setInterval(m_tickTimer.interval()+1);
147 qDebug() << "Tick interval set to " << m_tickTimer.interval();
148 break;
149 case Qt::Key_M:
150 m_tickTimer.setInterval(m_tickTimer.interval()-1);
151 qDebug() << "Tick interval set to " << m_tickTimer.interval();
152 break;
153 }
154 }
155 pKeyEvent->accept();
156 }
157}

203void GameCanvas::onTick() {
204 long long elapsedTime = m_lastUpdateTime.elapsed();
205
206 // On évite une division par zéro (peu probable, mais on sait jamais)
207 if (elapsedTime < 1)
208 elapsedTime = 1;
209
210 m_lastUpdateTime.start();
211
212 m_pGameCore->tick(elapsedTime);
213 currentScene()->tick(elapsedTime);
214
215 if (m_pDetailedInfosItem && m_pDetailedInfosItem->isVisible())
216 m_pDetailedInfosItem->setPlainText(QString("FPS : %1, Elapsed : %2ms, Tick duration : %3ms")
217 .arg(1000/elapsedTime)
218 .arg(elapsedTime)
219 .arg(m_lastUpdateTime.elapsed()));
220
221 if (m_keepTicking)
222 m_tickTimer.start();
223}

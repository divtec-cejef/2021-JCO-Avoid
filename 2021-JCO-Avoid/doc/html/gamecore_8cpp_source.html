168void GameCore::keyPressed(int key) {
169 emit notifyKeyPressed(key);
170
171 switch (key) {
172 case Qt::Key_Space:
173 reduceWalkingManOpacity();
174 break;
175 case Qt::Key_Q:
176 removeBall();
177 break;
178 case Qt::Key_X:
179 swapScenes();
180 break;
181 case Qt::Key_I:
182 displaySpriteDebugInfos();
183 break;
184 case Qt::Key_M:
185 toggleSpinningWheel();
186 break;
187 case Qt::Key_T:
188 setupHeadToShoot();
189 break;
190
191 case Qt::Key_D:
192 m_pPlayer->setX(m_pPlayer->x() + 20);
193 break;
194 case Qt::Key_A:
195 m_pPlayer->setX(m_pPlayer->x() - 20);
196 break;
197 case Qt::Key_N:
198 m_pPlayer->deleteLater();
199 onSpriteDestroyed(m_pPlayer);
200 m_pPlayer = nullptr;
201 break;

270void GameCore::mouseButtonPressed(QPointF mousePosition, Qt::MouseButtons buttons) {

272
273 // Si le bouton droite est cliqué et que le clip n'a pas déjà démarré,
274 // on démarre le clip de la balle qui tombe et rebondit sur le sol.
275 if (buttons.testFlag(Qt::RightButton) && !m_footballClipRunning) {
276 m_footballClipRunning = true;
277 createFootBallClip(mousePosition);
278 } else if (buttons.testFlag(Qt::LeftButton)) {
279
280 // bouton gauche cliqué : on vérifie si le sprite en dessous peut être shooté.
281 Sprite* pTargetSprite = m_pScene->spriteAt(mousePosition);
282 if (pTargetSprite->data(SPRITE_TYPE_KEY).toString() == HEADSHOOT_TYPE) {
283 // Le sprite est une tête à tirer : elle est détruite et explose
284 pTargetSprite->deleteLater();

473void GameCore::createFootBallClip(QPointF position) {
474 // Créé la balle de foot
475 Sprite* pFootBall = new Sprite(GameFramework::imagesPath() + "football.png");
476 pFootBall->setScale(0.1);
477 pFootBall->setZValue(0.2);
478
479 // Créé l'herbe
480 Sprite* pGrass = new Sprite(GameFramework::imagesPath() + "grass.png");
481 pGrass->setScale(0.1);
482 pGrass->setZValue(0.4);
483
484 // Met en place le clip
485 QRectF ClipRect(position, QSizeF(pGrass->width(), pGrass->width()*2));
486
487 QPixmap SkyPixmap(static_cast<int>(ClipRect.width()), static_cast<int>(ClipRect.height()));
488 SkyPixmap.fill(Qt::cyan);
489 Sprite* pSky = new Sprite(SkyPixmap);
490 pSky->setPos(ClipRect.topLeft());
491
492 QPointF BallStartPos(ClipRect.center().x() - pFootBall->width()/2, ClipRect.top());
493
494 pFootBall->setPos(BallStartPos);
495 pGrass->setPos(position.x(), ClipRect.bottom() - pGrass->height()/2);
496
497 m_footballClipSpriteList << pFootBall << pGrass << pSky;
498
499 QPropertyAnimation* pBallFallDownAnimation = new QPropertyAnimation(pFootBall, "y"); // La propriété à animer est y (la position y)
500 pBallFallDownAnimation->setDuration(1500);
501 pBallFallDownAnimation->setStartValue(BallStartPos.y());
502 pBallFallDownAnimation->setEndValue(BallStartPos.y() + ClipRect.height() - pFootBall->height());
503 pBallFallDownAnimation->setEasingCurve(QEasingCurve::OutBounce);
504
505 QParallelAnimationGroup* pAllVanishAnimation = new QParallelAnimationGroup;
506 pAllVanishAnimation->addAnimation(createVanishAnimation(pFootBall));
507 pAllVanishAnimation->addAnimation(createVanishAnimation(pGrass));
508 pAllVanishAnimation->addAnimation(createVanishAnimation(pSky));
509
510 QSequentialAnimationGroup* pClipAnimation = new QSequentialAnimationGroup;
511 pClipAnimation->addAnimation(pBallFallDownAnimation);
512 pClipAnimation->addAnimation(pAllVanishAnimation);
513
514 m_pScene->addSpriteToScene(pFootBall);
515 m_pScene->addSpriteToScene(pGrass);
516 m_pScene->addSpriteToScene(pSky);
517
518 connect(pClipAnimation, SIGNAL(finished()), this, SLOT(onClipVanishFinished()));
519
520 pClipAnimation->start(QAbstractAnimation::DeleteWhenStopped);
521}

570void GameCore::createExplosion(QPointF pos) {
571 Sprite* pExplosion = new Sprite;
572
573 // Chargement de la spritesheet
574 QImage spriteSheet(GameFramework::imagesPath() + "explosion.png");
575
576 // Découpage de la spritesheet
577 for (int frameIndex = 0; frameIndex < EXPLOSION_SHEET_COUNT; frameIndex++) {
578 QImage sprite = spriteSheet.copy((frameIndex % EXPLOSION_SHEET_COLUMN_COUNT) * EXPLOSION_SHEET_WIDTH,
579 (frameIndex / EXPLOSION_SHEET_COLUMN_COUNT) * EXPLOSION_SHEET_HEIGHT,
580 EXPLOSION_SHEET_WIDTH, EXPLOSION_SHEET_HEIGHT);
581
582 pExplosion->addAnimationFrame(QPixmap::fromImage(sprite.scaled(EXPLOSION_SHEET_WIDTH * 0.5,
583 EXPLOSION_SHEET_HEIGHT * 0.5,
584 Qt::IgnoreAspectRatio,
585 Qt::SmoothTransformation)));
586 }
587
588 pExplosion->setAnimationSpeed(25);
589 pExplosion->setPos(pos);
590
591 // Faire en sorte qu'à la fin de l'animation, le sprite est automatiquement détruit.
592 pExplosion->setEmitSignalEndOfAnimationEnabled(true);
593 connect(pExplosion, &Sprite::animationFinished, pExplosion, &Sprite::deleteLater);
594
595 // Démarrage de l'animation
596 m_pScene->addSpriteToScene(pExplosion);
597 pExplosion->startAnimation();
598

46void ObjetTickHandler::tick(long long elapsedTimeInMilliseconds) {
47 // Création d'un vecteur de déplacement du sprite.
48 QPointF spriteMovement(0, s_vitesseObtsacle);
49 // Détermine la prochaine position du sprite
50 QRectF nextSpriteRect = m_pParentSprite->globalBoundingBox().translated(spriteMovement);
51 // Récupère tous les sprites de la scène que toucherait ce sprite à sa prochaine position
52 auto collidingSprites = m_pParentSprite->parentScene()->collidingSprites(nextSpriteRect);
53 // Supprimer le sprite lui-même, qui collisionne toujours avec sa boundingbox
54 collidingSprites.removeAll(m_pParentSprite);
55
56 // Si la prochaine position du sprite n'est pas comprise au sein de la scène,
57 // le sprite n’est pas déplacé et est détruit
58 if (!m_pParentSprite->parentScene()->isInsideScene(nextSpriteRect))
59 m_pParentSprite->deleteLater();
60 else
61 // Si le sprite ne sort pas de la scène, on le déplace
62 // (en lui appliquant le vecteur de déplacement)
63 m_pParentSprite->setPos(m_pParentSprite->pos() + spriteMovement);
64
65}

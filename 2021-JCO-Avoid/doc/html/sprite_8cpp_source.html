242void Sprite::onNextAnimationFrame() {
243 if (m_animationList[m_currentAnimationIndex].isEmpty()) {
244 m_currentAnimationFrame = NO_CURRENT_FRAME;
245 return;
246 }
247
248 int PreviousAnimationFrame = m_currentAnimationFrame;
249 ++m_currentAnimationFrame;
250 if (m_currentAnimationFrame >= m_animationList[m_currentAnimationIndex].count()) {
251 if (m_emitSignalEOA)
252 emit animationFinished();
253 m_currentAnimationFrame = 0;
254 }
255 if (PreviousAnimationFrame != m_currentAnimationFrame) {
256 setPixmap(m_animationList[m_currentAnimationIndex][m_currentAnimationFrame]);
257 update();
258 }

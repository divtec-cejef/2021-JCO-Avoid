37void SpinningPinwheel::configureAnimation() {
38 // Chargement de la spritesheet
39 QImage spriteSheet(GameFramework::imagesPath() + "spinningwheel.png");
40
41 // Découpage de la spritesheet
42 for (int frameIndex = 0; frameIndex < FRAME_COUNT; frameIndex++) {
43 QImage sprite = spriteSheet.copy((frameIndex % COLUMN_COUNT) * FRAME_WIDTH,
44 (frameIndex / COLUMN_COUNT) * FRAME_HEIGHT,
45 FRAME_WIDTH, FRAME_HEIGHT);
46
47 this->addAnimationFrame(QPixmap::fromImage(sprite.scaled(FRAME_WIDTH * SCALE_RATIO,
48 FRAME_HEIGHT * SCALE_RATIO,
49 Qt::IgnoreAspectRatio,
50 Qt::SmoothTransformation)));
51 }
52
53 this->setAnimationSpeed(15);
54 updateAnimationState();
55}

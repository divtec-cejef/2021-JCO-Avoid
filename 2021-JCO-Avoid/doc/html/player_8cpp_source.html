35void Player::tick(long long elapsedTimeInMilliseconds) {

37 // Calcul de la distance parcourue par la balle, selon sa vitesse et le temps écoulé.
38 QPointF ballDistance = elapsedTimeInMilliseconds * m_ballVelocity / 1000.;

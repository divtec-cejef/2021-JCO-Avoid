112void GameCore::setupObstacle(){
113 nbGen=rand()%LARGEUR_MAX+LARGEUR_MINIMUM; //génère un chiffre aléatoire entre la largeur maximal et minimal de l'écran
114
115 nbGenObstacle = rand()%(INDEX_IMAGE_MAX - INDEX_IMAGE_MIN) + INDEX_IMAGE_MIN;
116 //Si le jeux n'est pas terminé continue de créer des obstacles et des bonus
117 if(jeuTermine == false){
118 //Si le nombre d'obstacle apparu après un bonus est égal à l'interval de l'apparition des bonus un bonus est générer
119 if(nombreObstacle == INTERVAL_APPARITION_BONUS){
120 setupBonus();
121 nombreObstacle = 0;
122 //sinon génére un obstacle à une position x aléatoire
123 }else{
124 nombreObstacle++;
125 Sprite* pObstacle = new Sprite(QString(GameFramework::imagesPath() + "obstacle/%0.png").arg(nbGenObstacle));
126 pObstacle->setPos(nbGen,0);
127 pObstacle->setScale(0.5);
128
129 // Déplace le sprite vers le bas.
130 // Si une collision à lieu, le tickhandler se charge
131 // de détruire son sprite.
132 pTickHandler = new ObjetTickHandler;
133 pTickHandler->setDestroyOnCollisionEnabled(true);
134 pObstacle->setTickHandler(pTickHandler);
135
136 m_pScene->addSpriteToScene(pObstacle);
137
138 pObstacle->registerForTick();
139 }
140 }

212void GameCore::updateProgressBar() {
213
214 if(jeuTermine == false) {
215 double newWidth = PROGRESSBAR_WIDTH / 100 * pProgressBar->getProgressbarProcent();
216 double screenXCenter = m_pScene->width() / 2;
217 m_pProgressBarFill->setRect(screenXCenter - PROGRESSBAR_WIDTH/2, 10, newWidth, 50);
218
219 //Si la barre de progression est rempli au plus du 2/3 la barre est verte
220 if (pProgressBar->getProgressbarProcent() > 100/3*2) {
221 m_pProgressBarFill->setBrush(QBrush(Qt::green));
222 //Si la barre de progression est rempli au plus du 1/3 la barre est orange
223 } else if (pProgressBar->getProgressbarProcent() > 100 / 3) {
224 m_pProgressBarFill->setBrush(QBrush(Qt::yellow));
225 } else {
226 //Sinon la barre est rouge
227 m_pProgressBarFill->setBrush(QBrush(Qt::red));
228 }
229 //Si la barre de progression atteint les 0% le jeux est arrêté
230 if (pProgressBar->getProgressbarProcent() <= 0) {
231 stopGame();
232 }
233 }

441void GameCore::mouseButtonPressed(QPointF mousePosition, Qt::MouseButtons buttons) {
442 emit notifyMouseButtonPressed(mousePosition, buttons);
443
444 // Si le bouton droite est cliqué et que le clip n'a pas déjà démarré,
445 // on démarre le clip de la balle qui tombe et rebondit sur le sol.
446 if (buttons.testFlag(Qt::LeftButton)) {
447 if (jeuTermine) {
448 // bouton gauche cliqué : on vérifie si le sprite en dessous peut être shooté.
449 Sprite* pTargetSprite = m_pScene->spriteAt(mousePosition);
450 if (pTargetSprite == pBouton) {
451 // Le sprite est une tête à tirer : elle est détruite et explose
452 pTickHandler->s_vitesseObtsacle = 7;
453 restartGame();
454 }

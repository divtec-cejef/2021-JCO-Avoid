329void Sprite::onNextAnimationFrame() {
330 if (m_animationList[m_currentAnimationIndex].isEmpty()) {
331 m_currentAnimationFrame = NO_CURRENT_FRAME;
332 return;
333 }
334
335 int PreviousAnimationFrame = m_currentAnimationFrame;
336 ++m_currentAnimationFrame;
337 if (m_currentAnimationFrame >= m_animationList[m_currentAnimationIndex].count()) {
338 m_currentAnimationFrame = 0;
339 if (m_emitSignalEOA)
340 emit animationFinished();
341 }
342 if (PreviousAnimationFrame != m_currentAnimationFrame) {
343 setPixmap(m_animationList[m_currentAnimationIndex][m_currentAnimationFrame]);
344 update();
345 }
